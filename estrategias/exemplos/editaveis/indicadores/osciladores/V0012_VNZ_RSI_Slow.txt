descrição:
plota a media entre o RSI e A MEdia(um nivel como Ocilador)
}
input
  periods(6);
  smooth(14);
  price(close);
var
  MA_s              : float;
  r1,r2,r3,r4,r5,rr : float;
  MA_s2             : float;
begin
  r1 := mediaexp(periods,price);
  if price > r1 then
    r2 := price - r1
  else 
    r2 := 0;
  if price < r1 then
    R3 := r1 - price
  else 
    R3 := 0;
  MA_s := (r2 + (MA_s[1] * (smooth - 1))) / smooth;
  r4 := MA_s;
  MA_s2 := (r3 + (MA_s2[1] * (smooth - 1))) / smooth;
  r5 := MA_s2;
  if r5 = 0 then
    rr := 100
  else 
    rr := 100 - (100 / (1 + (r4 / r5)));
  plot(rr);
end;
