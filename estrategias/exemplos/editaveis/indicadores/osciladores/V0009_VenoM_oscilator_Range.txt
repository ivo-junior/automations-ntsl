input
MedRapida(5);
MedLenta(34);
var
mPrice,AO,AC:float;
begin
mPrice:= (high+low)/2;  //preço medio
 AO:= (media(MedRapida,mPrice)-media(MedLenta,mPrice)); //saldo do resultado das medias de 5 e 34
 AC:= AO-media(MedRapida,AO);
 plot(AO);
 plot4(AC);
 plot2(20);
 plot3(-20);
end;
